@name Drone-Control by Gooch (1.3 - Modified)
@inputs Pod:wirelink Cam:wirelink Base:entity 
@outputs RGB:vector Torque:number
@persist Torque
if(first()){Torque = 0} runOnTick(1) if(Base:isValid()){ Active = Pod["Active",number] Ang = Pod["Entity",entity]:driver():eyeAngles() W = Pod["W",number] A = Pod["A",number] S = Pod["S",number] D = Pod["D",number] Space = Pod["Space",number] Shift = Pod["Shift",number] Cam["Angle",angle] = Base:angles() Cam["Position",vector] = Base:pos()+vec(-80,-2.25,15):rotate(Base:angles()) Cam["Activated",number] = Pod["Active",number] Base:applyForce(Base:forward()*Base:mass()*Torque) Base:applyForce(Base:right()*(D-A)*Base:mass()*80) Base:applyForce(Base:up()*(Space-Shift)*Base:mass()*80) Base:applyForce(-Base:vel()*Base:mass()/7.5) Base:setMass(50000)  if(!Active){Torque = 0 Base:propGravity(0)}else{Base:propGravity(1) Base:applyForce(vec(0,0,17.3)*Base:mass())} V = Base:toLocal(rotationVector(quat(Ang)/quat(Base))+Base:pos()) Base:applyTorque((200*V - 25*Base:angVelVector())*Base:inertia()) }  if (W & Torque<180) {Torque = Torque+2} if( S&Torque>(-120) ){Torque = Torque - 15} if(!W & !S){ if(Torque>0){Torque = Torque - 1} if(Torque < 0){Torque = Torque + 5} } if(dupefinished()){reset()} RGB = hsv2rgb((1 * 360 / 51) + (curtime() * 100), 1, 1)
